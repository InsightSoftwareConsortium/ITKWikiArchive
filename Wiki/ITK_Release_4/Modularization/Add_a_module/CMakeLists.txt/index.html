If this module only contains template source codes (no "src" subdirectory): 

 project(ITKFoo)
  
 #itk_module_impl() takes care of the installation and export rule for the 
 #module ITKFoo
 itk_module_impl()


If this module generates a library (has a "src" subdirectory): 

  project(ITKFoo)
  
  set(ITKFoo_LIBRARIES ITKFoo) # need to specify the library names
  itk_module_impl()
